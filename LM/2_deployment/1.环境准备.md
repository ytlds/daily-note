

显卡：Nvidia 4090

软件环境见 dockerfile

```dockerfile
FROM nvidia/cuda:12.8.0-devel-ubuntu22.04

ENV DEBIAN_FRONTEND=noninteractive

RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    cmake \
    python3.10 \
    python3-pip \
    git \
    git-lfs \
    curl \
    wget \
    vim \
    less \
    tmux \
    tree \
    htop \
    zsh \
    openssh-server &&\
    rm -rf /var/lib/apt/lists/* && \
    ln -s /usr/bin/python3.10 /usr/bin/python && \
    curl -sS https://bootstrap.pypa.io/get-pip.py | python3.10 && \
    pip install --upgrade pip

# LM
RUN pip install --no-cache-dir torch==2.7.1 torchvision==0.22.1 torchaudio==2.7.1 --extra-index-url https://download.pytorch.org/whl/cu128 && \
    pip install --no-cache-dir sglang[all]==0.4.8

```
